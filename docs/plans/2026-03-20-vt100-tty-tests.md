``` go
package crap

import (
    "image/color"
    "strings"
    "testing"

    "github.com/tonistiigi/vt100"
)

// renderScreen feeds raw output bytes through a 120x24 VT100 emulator
// and returns the visible lines (trailing blank lines trimmed) and the
// full Format grid.
func renderScreen(raw []byte) ([]string, [][]vt100.Format) {
    term := vt100.NewVT100(24, 120)
    term.Write(raw)

    var lines []string
    for _, row := range term.Content {
        lines = append(lines, strings.TrimRight(string(row), " "))
    }

    // Trim trailing blank lines.
    for len(lines) > 0 && lines[len(lines)-1] == "" {
        lines = lines[:len(lines)-1]
    }

    return lines, term.Format
}

// findLine returns the index of the first line containing substr, or -1.
func findLine(lines []string, substr string) int {
    for i, line := range lines {
        if strings.Contains(line, substr) {
            return i
        }
    }
    return -1
}

// findLinePrefix returns the index of the first line whose trimmed content
// starts with prefix, or -1.
func findLinePrefix(lines []string, prefix string) int {
    for i, line := range lines {
        if strings.HasPrefix(strings.TrimLeft(line, " "), prefix) {
            return i
        }
    }
    return -1
}

// cellsHaveColor checks that every cell in formats[row][startCol:endCol]
// has the expected foreground color.
func cellsHaveColor(formats [][]vt100.Format, row, startCol, endCol int, fg color.RGBA) bool {
    if row >= len(formats) {
        return false
    }
    for col := startCol; col < endCol && col < len(formats[row]); col++ {
        if formats[row][col].Fg != fg {
            return false
        }
    }
    return true
}

// cellsAreDim checks that every cell in formats[row][startCol:endCol]
// has Intensity == Dim.
func cellsAreDim(formats [][]vt100.Format, row, startCol, endCol int) bool {
    if row >= len(formats) {
        return false
    }
    for col := startCol; col < endCol && col < len(formats[row]); col++ {
        if formats[row][col].Intensity != vt100.Dim {
            return false
        }
    }
    return true
}

func TestVTRenderScreenSmoke(t *testing.T) {
    raw := []byte("hello world\n")
    lines, _ := renderScreen(raw)
    if len(lines) == 0 {
        t.Fatal("expected at least one line")
    }
    if !strings.Contains(lines[0], "hello world") {
        t.Errorf("expected 'hello world', got %q", lines[0])
    }
}
```

``` go
import (
    "context"
    "bytes"
    "os"
    "path/filepath"
)

// writeTAPScript creates a temporary shell script that emits TAP-14 with
// subtests. Returns the script path. The script has no sleeps so tests
// complete instantly.
func writeTAPScript(t *testing.T) string {
    t.Helper()
    dir := t.TempDir()
    path := filepath.Join(dir, "emit-tap.sh")
    script := `#!/bin/sh
cat <<'EOF'
TAP version 14
1::3
# Subtest: compilation
    1::2
    ok 1 - compile main.rs
      ---
      output: |
        compiling main.rs
        compiling lib.rs
        linking binary
      ...
    ok 2 - compile tests
      ---
      output: |
        compiling test_parse.rs
        compiling test_format.rs
      ...
ok 1 - compilation
# Subtest: test suite
    1::3
    ok 1 - test_parse
      ---
      output: |
        running test_parse
        assertion passed: expected 42 got 42
      ...
    not ok 2 - test_format
      ---
      message: "assertion failed"
      severity: fail
      output: |
        running test_format
        FAIL: expected "hello" got "world"
      ...
    ok 3 - test_lint # skip no linter configured
ok 2 - test suite
ok 3 - cleanup
1::3
EOF
`
    os.WriteFile(path, []byte(script), 0o755)
    return path
}

func runTAPFixture(t *testing.T, color bool) ([]string, [][]vt100.Format) {
    t.Helper()
    script := writeTAPScript(t)
    var buf bytes.Buffer
    code := RunWithPTYReformat(
        context.Background(), script, nil, &buf, color,
        WithSpinner(false),
    )
    if code != 0 {
        t.Fatalf("expected exit code 0, got %d\nraw output:\n%s", code, buf.String())
    }
    return renderScreen(buf.Bytes())
}

func TestVTTAPVersionHeader(t *testing.T) {
    lines, _ := runTAPFixture(t, false)
    if len(lines) == 0 || lines[0] != "CRAP-2" {
        t.Errorf("expected first line 'CRAP-2', got %q", lines[0])
    }
}
```

### Task 5: No status line artifacts in final screen

**Files:** - Modify: `go-crap/ptyrun_vt_test.go`

**Step 1: Add status line artifact test**

The status line uses `#` prefix and cursor-up/erase sequences. After the command
finishes, no status lines should remain visible.

Append to `go-crap/ptyrun_vt_test.go`:

``` go
func TestVTTAPNoStatusLineArtifacts(t *testing.T) {
    lines, _ := runTAPFixture(t, true)
    for i, line := range lines {
        trimmed := strings.TrimLeft(line, " ")
        // Lines starting with # are TAP comments (e.g., "# Subtest:") and
        // YAML content — those are fine. Status line artifacts would be
        // orphaned "# " lines that don't belong to the TAP structure.
        // The key indicator: status lines contain raw output text after "# "
        // that duplicates a nearby TAP line (e.g., "# ok 1 - compilation").
        if strings.HasPrefix(trimmed, "# ok ") || strings.HasPrefix(trimmed, "# not ok ") {
            t.Errorf("line %d looks like a status line artifact: %q", i, line)
        }
        if strings.HasPrefix(trimmed, "# 1::") {
            t.Errorf("line %d looks like a status line artifact: %q", i, line)
        }
    }
}
```
